\chapter{2}{Algebra}
\subsection*{Räkneoperationer för reella tal}

\begin{task}{2.1 a)}
	Två alternativa lösningsmetoder:
	\[(x+3)(x-3)-(x+3)^2=\cancel{x^2}-9-(\cancel{x^2}+6x+9)=-6x-18\]
	eller
	\[(x+3)(x-3)-(x+3)^2=(x+3)((\cancel{x}-3)-(\cancel{x}+3))=-6(x+3)=-6x-18\]
	
	\ans $-6x-18$
\end{task}
